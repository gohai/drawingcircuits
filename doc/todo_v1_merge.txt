		$('.tool').on('click', function(e) {
			var p = $(this).prop('id').lastIndexOf('-');
			if (p === -1) {
				return;
			}
			$.pcb.tool($(this).prop('id').substr(p+1));
		});
		$('#tool-draw').on('click', function(e) {
			if (e.shiftKey || e.altKey) {
				var s = prompt('Set diameter (mm)', $.pcb.diameter());
				if (s === null) {
					return;
				}
				$.pcb.diameter(parseFloat(s));
			}
			if (e.altKey) {
				promptManualCoords();
			}
		});
		$('#tool-erase').on('click', function(e) {
			if (e.shiftKey || e.altKey) {
				var s = prompt('Set diameter (mm)', $.pcb.diameter());
				if (s === null) {
					return;
				}
				$.pcb.diameter(parseFloat(s));	
			}
			if (e.altKey) {
				promptManualCoords();
			}
		});
		$('#tool-drill').on('click', function(e) {
			if (e.shiftKey || e.altKey) {
				var s = prompt('Set diameter (mm)', $.pcb.diameter());
				if (s === null) {
					return;
				}
				$.pcb.diameter(parseFloat(s));	
			}
			if (e.altKey) {
				promptManualCoords();
			}
		});
		$('#menu-del').on('click', function(e) {
			$.pcb.remove();
		});
		$('#tool-parts').on('click', function(e) {
			$.pcb.showLibrary();
			return false;
		});
		$('#menu-zoomin').on('click', function(e) {
			$.pcb.zoom($.pcb.zoom()-0.25);
		});
		$('#menu-zoomout').on('click', function(e) {
			$.pcb.zoom($.pcb.zoom()+0.25);
		});
		$('#menu-ruler').on('click', function(e) {
			$.pcb.ruler(!$.pcb.ruler());
		});
		$('.menu-layer').on('click', function(e) {
			var p = $(this).prop('id').lastIndexOf('-');
			if (p === -1) {
				return;
			}
			$.pcb.layer($(this).prop('id').substr(p+1));
		});
		$('#menu-save').on('click', function(e) {
			if (e.shiftKey) {
				$.pcb.save(true);
			} else {
				$.pcb.save();
			}
			// visual feedback
			$(this).css('background-color', 'lightblue');
			var that = this;
			var checkSave = function() {
				if ($.pcb.requestDone()) {
					$(that).css('background-color', '');
				} else {
					setTimeout(checkSave, 100);
				}
			};
			// TODO: just checkSave()?
			setTimeout(checkSave, 100);
		});
		$('#menu-export').on('click', function(e) {
			window.location = options.baseUrl+'ajax.php?method=export&type=rml&board=5';
			//+options.board+'&rev='+options.rev;
		});
		$('#menu-import').on('click', function(e) {
			$.pcb.addPartToLibrary();
			// prevent the dialog from closing right away
			return false;
		});
		$('#menu-clear').on('click', function(e) {
			if (confirm("Clear board?")) {
				$.pcb.clear();
			}
		});


var promptManualCoords = function() {
		s = prompt('Set x coordinate (mm)', options.lastManualX);
		if (s === null) {
			return;
		}
		var x = parseFloat(s);
		s = prompt('Set y coordinate (mm)', options.lastManualY);
		if (s === null) {
			return;
		}
		var y = parseFloat(s);
		$.pcb.point(x, y);
		lastManualX = x;
		lastManualY = y;
	}
